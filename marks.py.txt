def calculate_result(marks):
    total = sum(marks)
    average = total / len(marks)

    if average >= 40:
        result = "Pass"
    else:
        result = "Fail"

    return total, average, result


name = input("Enter student name: ")

marks = []
for i in range(1, 6):
    m = int(input(f"Enter marks of subject {i}: "))
    marks.append(m)

total, avg, res = calculate_result(marks)

print("\n--- Student Report ---")
print("Name:", name)
print("Total Marks:", total)
print("Average:", avg)
print("Result:", res)
